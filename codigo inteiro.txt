				<script type="text/javascript" id="scheduling-listener">

//Global Vars
var trackingStep = 1; //this is used to keep the mutation observer from firing multiple events
var zipCode; //this zipcode can be used as a GTM variable and sent to GA4 as a parameter and custom dimension
    
    jQuery(document).ready(function($) {
        
        //Variables
        var target = document.querySelector('#root');
        var observer = new MutationObserver(function(mutations) { //this listens for changes on the page
            mutations.forEach(function(mutation) {
                scheduling_listener();
            });
        });
    
        // configuration of the observer
        var config = { characterData: true, subtree: true, childList: true };
    
        observer.observe(target, config);
    
        function scheduling_listener() {
    
                if ($('h1:contains("Welcome to Floor Coverings International")').length && trackingStep == 1) {
                //When the appointment system initiates
                console.log('step 1');
                trackingStep = 2;
                }
    
                if ($('h1:contains("Who You Are")').length && trackingStep == 2) {
                //The page with the lead form
                console.log('step 2');
                dataLayer.push({
                        'event': 'appointmentStart'
                    });
                zip_listener();
                trackingStep = 3;
                }
    
                if ($('h1:contains("book your appointment")').length && trackingStep == 3) {
                //The page with the appointment selection calendar
                console.log('step 3');
                dataLayer.push({
                        'event': 'appointmentLead'
                    });
                trackingStep = 4;
                }
          
                if ($('h1:contains("In-Home Consultation")').length && trackingStep == 4) {
                //The page with the appointment selection calendar
                console.log('step 4');
                trackingStep = 5;
                }
    
                if ($('h1:contains("Congratulations")').length && trackingStep == 5) {
                //The appointment collection page
                console.log('step 5');
                dataLayer.push({
                        'event': 'appointmentBooked'
                    });
                trackingStep = 6;
                }
    
            }
        function zip_listener() {
    
                $("input.submit-zip-field, input#zip").change(function(){
                    zipCode = $(this).val();
                    window.dataLayer.push({
                        'zipcode': zipCode
                    }); 
                });
    
        }
        
        scheduling_listener();
        zip_listener();
    
    });
    </script>

    This script can be placed into the web application itself or loaded via Google Tag Manager in a Custom HTML tag. 

    Here are the components that need to be added to Google Tag Manager to process the data:

    1. Variables

        - Zipcode / set the Zipcode variable to a dataLayer variable with the value of: zipCode
    
    2. Triggers

        - Appointment Start / Set as a custom event with the value of: appointmentStart
        - Appointment Lead / Set as a custom event with the value of: appointmentLead
        - Appointment Booked / Set as a custom event with the value of: appointmentBooked

    3. GA4 Event

        - Create GA4 event tags for each of the triggers to send your event name (e.g. appointment_start, appointment_lead, appointment_booked)
